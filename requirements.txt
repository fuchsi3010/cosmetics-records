# GUI Framework
# PyQt6 is a modern Python binding for Qt6, providing native-looking
# cross-platform GUI components

# allowing the app to match the system theme
darkdetect>=0.8.0

# FontAwesome Icons
# qtawesome provides FontAwesome and other icon fonts for PyQt6
# Used for navigation icons and other UI elements
qtawesome>=1.3.0
